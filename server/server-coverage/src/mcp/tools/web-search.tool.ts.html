import type { McpToolImplementation } from "../core/definitions.js";
import { mcpBadRequest, mcpInternalError } from "../core/errors.js";
 
export interface SearchResult {
  title: string;
  link: string;
  snippet: string;
}
 
type SearchProvider = "auto" | "duckduckgo" | "tavily";
 
const SEARCH_TIMEOUT_MS = 10_000;
 
const withTimeoutSignal = () => AbortSignal.timeout(SEARCH_TIMEOUT_MS);
 
const normalizeText = (value: string) =>
  value
    .replace(/<[^>]+>/g, " ")
    .replace(/&amp;/gi, "&")
    .replace(/&quot;/gi, '"')
    .replace(/&#39;/gi, "'")
    .replace(/&lt;/gi, "<")
    .replace(/&gt;/gi, ">")
    .replace(/\s+/g, " ")
    .trim();
 
const parseDuckDuckGoResults = (html: string, maxResults: number): SearchResult[] => {
  const results: SearchResult[] = [];
  const linkMatches = html.matchAll(
    /<a[^>]+class="[^"]*(?:result-link|result__a)[^"]*"[^>]+href="([^"]+)"[^>]*>([\s\S]*?)<\/a>/gi,
  );
 
  for (const match of linkMatches) {
    if (results.length >= maxResults) {
      break;
    }
 
    const title = normalizeText(match[2] ?? "");
    if (!title) {
      continue;
    }
 
    results.push({
      title,
      link: match[1] ?? "",
      snippet: "",
    });
  }
 
  const snippetMatches = html.matchAll(
    /<(?:td|div|a)[^>]+class="[^"]*(?:result-snippet|result__snippet)[^"]*"[^>]*>([\s\S]*?)<\/(?:td|div|a)>/gi,
  );
  let index = 0;
 
  for (const match of snippetMatches) {
    if (index >= results.length) {
      break;
    }
 
    results[index].snippet = normalizeText(match[1] ?? "");
    index += 1;
  }
 
  return results;
};
 
const resolveTavilyApiKey = (args: Record<string, unknown>) =>
  typeof args.apiKey === "string" && args.apiKey.trim()
    ? args.apiKey.trim()
    : (process.env.TAVILY_API_KEY ?? "").trim();
 
const fetchTavilySearch = async (
  query: string,
  maxResults: number,
  apiKey: string,
): Promise<SearchResult[]> => {
  if (!apiKey) {
    throw mcpInternalError("Tavily apiKey is required");
  }
 
  const response = await fetch("https://api.tavily.com/search", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    signal: withTimeoutSignal(),
    body: JSON.stringify({
      api_key: apiKey,
      query,
      search_depth: "basic",
      max_results: maxResults,
    }),
  });
 
  if (!response.ok) {
    throw mcpInternalError(`Tavily search failed: ${response.status}`);
  }
 
  const data = (await response.json()) as {
    results?: Array<{ title?: string; url?: string; content?: string }>;
  };
  return (data.results ?? []).map((item) => ({
    title: item.title ?? "",
    link: item.url ?? "",
    snippet: item.content ?? "",
  }));
};
 
const fetchDuckDuckGoSearch = async (query: string, maxResults: number): Promise<SearchResult[]> => {
  const urls = [
    `https://lite.duckduckgo.com/lite/?q=${encodeURIComponent(query)}`,
    `https://html.duckduckgo.com/html/?q=${encodeURIComponent(query)}`,
  ];
  const failures: string[] = [];
 
  for (const url of urls) {
    try {
      const response = await fetch(url, {
        headers: {
          "User-Agent": "Mozilla/5.0",
        },
        signal: withTimeoutSignal(),
      });
 
      if (!response.ok) {
        failures.push(`${url} -> ${response.status}`);
        continue;
      }
 
      const html = await response.text();
      const results = parseDuckDuckGoResults(html, maxResults);
      if (results.length > 0) {
        return results;
      }
 
      failures.push(`${url} -> no results parsed`);
    } catch (error) {
      failures.push(
        `${url} -> ${error instanceof Error ? error.message : String(error)}`,
      );
    }
  }
 
  throw mcpInternalError(
    `DuckDuckGo search failed (${failures.join("; ") || "no reachable endpoint"})`,
  );
};
 
const resolveProviderOrder = (
  provider: SearchProvider,
  tavilyApiKey: string,
): Array<Exclude<SearchProvider, "auto">> => {
  if (provider === "tavily") {
    return ["tavily"];
  }
 
  if (provider === "duckduckgo") {
    return ["duckduckgo"];
  }
 
  return tavilyApiKey
    ? ["tavily", "duckduckgo"]
    : ["duckduckgo", "tavily"];
};
 
export const webSearchTool: McpToolImplementation = {
  definition: {
    id: "web_search",
    title: "Web Search",
    description: "Search the public web through a configured provider.",
    domain: "web_search",
    mode: "sync",
    inputSchema: {
      type: "object",
      required: ["query"],
      properties: {
        query: { type: "string" },
        maxResults: { type: "number" },
        provider: {
          type: "string",
          enum: ["auto", "duckduckgo", "tavily"],
        },
        apiKey: { type: "string" },
      },
    },
    tags: ["search", "web"],
    capabilities: {
      sideEffect: "network",
      requiresApproval: false,
      networkAccess: true,
    },
  },
  execute: async (context) => {
    const query = typeof context.args.query === "string" ? context.args.query.trim() : "";
    if (!query) {
      throw mcpBadRequest("query is required");
    }
 
    const maxResults =
      typeof context.args.maxResults === "number" ? context.args.maxResults : 5;
    const tavilyApiKey = resolveTavilyApiKey(context.args);
    const provider =
      context.args.provider === "tavily" ||
      context.args.provider === "duckduckgo" ||
      context.args.provider === "auto"
        ? context.args.provider
        : "auto";
    const providerOrder = resolveProviderOrder(provider, tavilyApiKey);
 
    context.pushEvent({
      type: "invocation:progress",
      message: `Searching web with ${provider}`,
    });
 
    const failures: string[] = [];
    let resolvedProvider: Exclude<SearchProvider, "auto"> | null = null;
    let results: SearchResult[] = [];
 
    for (const candidate of providerOrder) {
      context.pushEvent({
        type: "invocation:progress",
        message: `Trying provider ${candidate}`,
      });
 
      try {
        results =
          candidate === "tavily"
            ? await fetchTavilySearch(query, maxResults, tavilyApiKey)
            : await fetchDuckDuckGoSearch(query, maxResults);
        resolvedProvider = candidate;
        break;
      } catch (error) {
        const message = error instanceof Error ? error.message : String(error);
        failures.push(`${candidate}: ${message}`);
        context.pushEvent({
          type: "invocation:progress",
          message: `Provider ${candidate} failed, trying next candidate`,
        });
      }
    }
 
    if (!resolvedProvider) {
      throw mcpInternalError(
        `All search providers failed (${failures.join("; ") || "unknown error"})`,
      );
    }
 
    context.addArtifact({
      kind: "search-results",
      title: `Search results for ${query}`,
      data: results,
      metadata: {
        query,
        provider: resolvedProvider,
        requestedProvider: provider,
        fallbackAttempts: failures,
      },
    });
 
    return {
      result: {
        query,
        provider: resolvedProvider,
        requestedProvider: provider,
        results,
        fallbackAttempts: failures,
      },
    };
  },
};
